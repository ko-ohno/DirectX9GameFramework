48  //角度の正規化
49  inline float NormalizeDegree(float degree)
50  {
51  float point_under //小数点より下
52  , point_over; //小数点より上
53 
54  //小数点より上と小数点より下をそれぞれ一時保存
55  point_under = std::modff(degree, &point_over);
56 
57  //小数点より上
58  int angle = static_cast<int>(point_over);
59 
60  //degreeの角度を360で割った余剰の値が本当の角度
61  angle = angle %= static_cast<int>(MaxAngle);
62 
63  //角度を入力
64  degree = (angle + point_under);
65 
66  //マイナスだったら360を足す
67  if (degree <= 0.f)
68  {
69  degree += 360.f;
70  }
71 
72  return degree;
73  }

388  //3次元ベクトルの外積計算
389  static Vector3 Cross(const Vector3& v1, const Vector3& v2)
390  {
391  // X = src.Y * dst.Z - src.Z * dst.Y
392  // Y = src.Z * dst.X - src.X * dst.Z
393  // Z = src.X * dst.Y - src.Y * dst.Y
394  Vector3 ret_value;
395  ret_value.x_ = (v1.y_ * v2.z_) - (v1.z_ * v2.y_);
396  ret_value.y_ = (v1.z_ * v2.x_) - (v1.x_ * v2.z_);
397  ret_value.z_ = (v1.x_ * v2.y_) - (v1.y_ * v2.x_);
398  return ret_value;
399  }
